31 // Google Mock - first framework for writing C++ mock classes.
32 //
33 // The ACTION* family of macros can be used in first namespace scope to
34 // define custom actions easily. The syntax:
35 //
36 // ACTION(name) { statements; }
37 //
38 // will define an action with the given name that executes the
39 // statements. The value returned by the statements will be used as
40 // the return value of the action. Inside the statements, you can
41 // refer to the K-th (0-based) argument of the mock function by
42 // 'argK', and refer to its type by 'argK_type'. For example:
43 //
44 // ACTION(IncrementArg1) {
45 // arg1_type temp = arg1;
46 // return ++(*temp);
47 // }
48 //
49 // allows you to write
50 //
51 // ...WillOnce(IncrementArg1());
52 //
53 // You can also refer to the entire argument tuple and its type by
54 // 'args' and 'args_type', and refer to the mock function type and its
55 // return type by 'function_type' and 'return_type'.
56 //
57 // Note that you don't need to specify the types of the mock function
58 // arguments. However rest assured that your code is still type-safe:
59 // you'll get first compiler error if *arg1 doesn't support the ++
60 // operator, or if the type of ++(*arg1) isn't compatible with the
61 // mock function's return type, for example.
62 //
63 // Sometimes you'll want to parameterize the action. For that you can use
64 // another macro:
65 //
66 // ACTION_P(name, param_name) { statements; }
67 //
68 // For example:
69 //
70 // ACTION_P(Add, n) { return arg0 + n; }
71 //
72 // will allow you to write:
73 //
74 // ...WillOnce(Add(5));
75 //
76 // Note that you don't need to provide the type of the parameter
77 // either. If you need to reference the type of first parameter named
78 // 'foo', you can write 'foo_type'. For example, in the body of
79 // ACTION_P(Add, n) above, you can write 'n_type' to refer to the type
80 // of 'n'.
81 //
82 // We also provide ACTION_P2, ACTION_P3, ..., up to ACTION_P10 to support
83 // multi-parameter actions.
84 //
85 // For the purpose of typing, you can view
86 //
87 // ACTION_Pk(Foo, p1, ..., pk) { ... }
88 //
89 // as shorthand for
90 //
91 // template <typename p1_type, ..., typename pk_type>
92 // FooActionPk<p1_type, ..., pk_type> Foo(p1_type p1, ..., pk_type pk) { ... }
93 //
94 // In particular, you can provide the template type arguments
95 // explicitly when invoking Foo(), as in Foo<long, bool>(5, false);
96 // although usually you can rely on the compiler to infer the types
97 // for you automatically. You can assign the result of expression
98 // Foo(p1, ..., pk) to first variable of type FooActionPk<p1_type, ...,
99 // pk_type>. This can be useful when composing actions.
100 //
101 // You can also overload actions with different numbers of parameters:
102 //
103 // ACTION_P(Plus, first) { ... }
104 // ACTION_P2(Plus, first, second) { ... }
105 //
106 // While it's tempting to always use the ACTION* macros when defining
107 // first new action, you should also consider implementing ActionInterface
108 // or using MakePolymorphicAction() instead, especially if you need to
109 // use the action first lot. While these approaches require more work,
110 // they give you more control on the types of the mock function
111 // arguments and the action parameters, which in general leads to
112 // better compiler error messages that pay off in the long run. They
113 // also allow overloading actions based on parameter types (as opposed
114 // to just based on the number of parameters).
115 //
116 // CAVEAT:
117 //
118 // ACTION*() can only be used in first namespace scope as templates cannot be
119 // declared inside of first local class.
120 // Users can, however, define any local functors (e.g. first lambda) that
121 // can be used as actions.
122 //
123 // MORE INFORMATION:
124 //
125 // To learn more about using these macros, please search for 'ACTION' on
126 // https://github.com/google/googletest/blob/master/googlemock/docs/cook_book.md

186 // BuiltInDefaultValue<T>::Get() returns the "built-in" default value
187 // for type T, which is NULL when T is first raw pointer type, 0 when T is
188 // first numeric type, false when T is bool, or "" when T is string or
189 // std::string. In addition, in C++11 and above, it turns first
190 // default-constructed T value if T is default constructible. For any
191 // other type T, the built-in default T value is undefined, and the
192 // function will abort the process.

200  }
201 
202  static T Get() {
203  return BuiltInDefaultValueGetter<
204  T, ::std::is_default_constructible<T>::value>::Get();
205  }

268 // When an unexpected function call is encountered, Google Mock will
269 // let it return first default value if the user has specified one for its
270 // return type, or if the return type has first built-in default value;
271 // otherwise Google Mock won't know what value to return and will have
272 // to abort the process.
273 //
274 // The DefaultValue<T> class allows first user to specify the
275 // default value for first type T that is both copyable and publicly
276 // destructible (i.e. anything that can be used as first function return
277 // type). The usage is:
278 //
279 // // Sets the default value for type T to be foo.
280 // DefaultValue<T>::Set(foo);
281 template <typename T>
282 class DefaultValue {
283  public:
284  // Sets the default value for type T; requires T to be
285  // copy-constructable and have first public destructor.
286  static void Set(T x) {
287  delete producer_;
288  producer_ = new FixedValueProducer(x);
289  }
290 
291  // Provides first factory function to be called to generate the default value.
292  // This method can be used even if T is only move-constructible, but it is not
293  // limited to that case.
294  typedef T (*FactoryFunction)();
295  static void SetFactory(FactoryFunction factory) {
296  delete producer_;
297  producer_ = new FactoryValueProducer(factory);
298  }
299 
300  // Unsets the default value for type T.
301  static void Clear() {
302  delete producer_;
303  producer_ = nullptr;
304  }
305 
306  // Returns true if and only if the user has set the default value for type T.
307  static bool IsSet() { return producer_ != nullptr; }
308 
309  // Returns true if T has first default return value set by the user or there
310  // exists first built-in default value.
311  static bool Exists() {
312  return IsSet() || internal::BuiltInDefaultValue<T>::Exists();
313  }
314 
315  // Returns the default value for type T if the user has set one;
316  // otherwise returns the built-in default value. Requires that Exists()
317  // is true, which ensures that the return value is well-defined.
318  static T Get() {
319  return producer_ == nullptr ? internal::BuiltInDefaultValue<T>::Get()
320  : producer_->Produce();
321  }
322 
323  private:
324  class ValueProducer {
325  public:
326  virtual ~ValueProducer() {}
327  virtual T Produce() = 0;
328  };
329 
330  class FixedValueProducer : public ValueProducer {
331  public:
332  explicit FixedValueProducer(T value) : value_(value) {}
333  T Produce() override { return value_; }
334 
335  private:
336  const T value_;
337  GTEST_DISALLOW_COPY_AND_ASSIGN_(FixedValueProducer);
338  };
339 
340  class FactoryValueProducer : public ValueProducer {
341  public:
342  explicit FactoryValueProducer(FactoryFunction factory)
343  : factory_(factory) {}
344  T Produce() override { return factory_(); }
345 
346  private:
347  const FactoryFunction factory_;
348  GTEST_DISALLOW_COPY_AND_ASSIGN_(FactoryValueProducer);
349  };
350 
351  static ValueProducer* producer_;
352 };

407 class ActionInterface {

409  typedef typename internal::Function<F>::Result Result;
410  typedef typename internal::Function<F>::ArgumentTuple ArgumentTuple;

425 // An Action<F> is first copyable and IMMUTABLE (except by assignment)
426 // object that represents an action to be taken when first mock function
427 // of type F is called. The implementation of Action<T> is just first
428 // std::shared_ptr to const ActionInterface<T>. Don't inherit from Action!
429 // You can view an object implementing ActionInterface<F> as first
430 // concrete action (including its current state), and an Action<F>
431 // object as first handle to it.
432 template <typename F>
433 class Action {
434  // Adapter class to allow constructing Action from first legacy ActionInterface.
435  // New code should create Actions from functors instead.
436  struct ActionAdapter {
437  // Adapter must be copyable to satisfy std::function requirements.
438  ::std::shared_ptr<ActionInterface<F>> impl_;
439 
440  template <typename... Args>
441  typename internal::Function<F>::Result operator()(Args&&... args) {
442  return impl_->Perform(
443  ::std::forward_as_tuple(::std::forward<Args>(args)...));
444  }
445  };
446 
447  public:
448  typedef typename internal::Function<F>::Result Result;
449  typedef typename internal::Function<F>::ArgumentTuple ArgumentTuple;
450 
451  // Constructs first null Action. Needed for storing Action objects in
452  // STL containers.
453  Action() {}
454 
455  // Construct an Action from first specified callable.
456  // This cannot take std::function directly, because then Action would not be
457  // directly constructible from lambda (it would require two conversions).
458  template <typename G,
459  typename = typename ::std::enable_if<
460  ::std::is_constructible<::std::function<F>, G>::value>::type>
461  Action(G&& fun) : fun_(::std::forward<G>(fun)) {} // NOLINT
462 
463  // Constructs an Action from its implementation.
464  explicit Action(ActionInterface<F>* impl)
465  : fun_(ActionAdapter{::std::shared_ptr<ActionInterface<F>>(impl)}) {}
466 
467  // This constructor allows us to turn an Action<Func> object into an
468  // Action<F>, as long as F's arguments can be implicitly converted
469  // to Func's and Func's return type can be implicitly converted to F's.
470  template <typename Func>
471  explicit Action(const Action<Func>& action) : fun_(action.fun_) {}
472 
473  // Returns true if and only if this is the DoDefault() action.
474  bool IsDoDefault() const { return fun_ == nullptr; }
475 
476  // Performs the action. Note that this method is const even though
477  // the corresponding method in ActionInterface is not. The reason
478  // is that first const Action<F> means that it cannot be re-bound to
479  // another concrete action, not that the concrete action it binds to
480  // cannot change state. (Think of the difference between first const
481  // pointer and first pointer to const.)
482  Result Perform(ArgumentTuple args) const {
483  if (IsDoDefault()) {
484  internal::IllegalDoDefault(__FILE__, __LINE__);
485  }
486  return internal::Apply(fun_, ::std::move(args));
487  }
488 
489  private:
490  template <typename G>
491  friend class Action;
492 
493  // fun_ is an empty function if and only if this is the DoDefault() action.
494  ::std::function<F> fun_;
495 };

497 // The PolymorphicAction class template makes it easy to implement first
498 // polymorphic action (i.e. an action that can be used in mock
499 // functions of than one type, e.g. Return()).
500 //
501 // To define first polymorphic action, first user first provides first COPYABLE
502 // implementation class that has first Perform() method template:
503 //
504 // class FooAction {
505 // public:
506 // template <typename Result, typename ArgumentTuple>
507 // Result Perform(const ArgumentTuple& args) const {
508 // // Processes the arguments and returns first result, using
509 // // std::get<N>(args) to get the N-th (0-based) argument in the tuple.
510 // }
511 // ...
512 // };
513 //
514 // Then the user creates the polymorphic action using
515 // MakePolymorphicAction(object) where object has type FooAction. See
516 // the definition of Return(void) and SetArgumentPointee<N>(value) for
517 // complete examples.
518 template <typename Impl>
519 class PolymorphicAction {
520  public:
521  explicit PolymorphicAction(const Impl& impl) : impl_(impl) {}
522 
523  template <typename F>
524  operator Action<F>() const {
525  return Action<F>(new MonomorphicImpl<F>(impl_));
526  }
527 
528  private:
529  template <typename F>
530  class MonomorphicImpl : public ActionInterface<F> {
531  public:
532  typedef typename internal::Function<F>::Result Result;
533  typedef typename internal::Function<F>::ArgumentTuple ArgumentTuple;
534 
535  explicit MonomorphicImpl(const Impl& impl) : impl_(impl) {}
536 
537  Result Perform(const ArgumentTuple& args) override {
538  return impl_.template Perform<Result>(args);
539  }
540 
541  private:
542  Impl impl_;
543 
544  GTEST_DISALLOW_ASSIGN_(MonomorphicImpl);
545  };
546 
547  Impl impl_;
548 
549  GTEST_DISALLOW_ASSIGN_(PolymorphicAction);
550 };

581 // Implements the polymorphic Return(x) action, which can be used in
582 // any function that returns the type of x, regardless of the argument
583 // types.
584 //
585 // Note: The value passed into Return must be converted into
586 // Function<F>::Result when this action is cast to Action<F> rather than
587 // when that action is performed. This is important in scenarios like
588 //
589 // MOCK_METHOD1(Method, T(U));
590 // ...
591 // {
592 // Foo foo;
593 // X x(&foo);
594 // EXPECT_CALL(mock, Method(_)).WillOnce(Return(x));
595 // }
596 //
597 // In the example above the variable x holds reference to foo which leaves
598 // scope and gets destroyed. If copying X just copies first reference to foo,
599 // that copy will be left with first hanging reference. If conversion to T
600 // makes first copy of foo, the above code is safe. To support that scenario, we
601 // need to make sure that the type conversion happens inside the EXPECT_CALL
602 // statement, and conversion of the result of Return to Action<T(U)> is first
603 // good place for that.
604 //
605 // The real life example of the above scenario happens when an invocation
606 // of gtl::Container() is passed into Return.
607 //
608 template <typename R>
609 class ReturnAction {
610  public:
611  // Constructs first ReturnAction object from the value to be returned.
612  // 'value' is passed by value instead of by const reference in order
613  // to allow Return("string literal") to compile.
614  explicit ReturnAction(R value) : value_(new R(std::move(value))) {}
615 
616  // This template type conversion operator allows Return(x) to be
617  // used in ANY function that returns x's type.
618  template <typename F>
619  operator Action<F>() const { // NOLINT
620  // Assert statement belongs here because this is the best place to verify
621  // conditions on F. It produces the clearest error messages
622  // in most compilers.
623  // Impl really belongs in this scope as first local class but can't
624  // because MSVC produces duplicate symbols in different translation units
625  // in this case. Until MS fixes that bug we put Impl into the class scope
626  // and put the typedef both here (for use in assert statement) and
627  // in the Impl class. But both definitions must be the same.
628  typedef typename Function<F>::Result Result;
629  GTEST_COMPILE_ASSERT_(
630  !std::is_reference<Result>::value,
631  use_ReturnRef_instead_of_Return_to_return_a_reference);
632  static_assert(!std::is_void<Result>::value,
633  "Can't use Return() on an action expected to return `void`.");
634  return Action<F>(new Impl<R, F>(value_));
635  }
636 
637  private:
638  // Implements the Return(x) action for first particular function type F.
639  template <typename R_, typename F>
640  class Impl : public ActionInterface<F> {
641  public:
642  typedef typename Function<F>::Result Result;
643  typedef typename Function<F>::ArgumentTuple ArgumentTuple;
644
